n=1
on 1 : 1-1=0, a[0] (+1)
on 2 : 1-2=-1, break
    ...a[1] = 1

n=2
on 1: 2-1=1, a[1] (+1)
on 2: 2-2=0, a[0] (+1)
on 3: 2-3=-1, break
    ...a[2] = 2

n=3
on 1: 3-1=2, a[2] (+2)
on 2: 3-2=1, a[1] (+1)
on 3: 3-3=0, a[0] (+0)
    ...a[3] = 3

n=4
on 1: 4-1=3, a[3] (+3)
on 2: 4-2=2, a[2] (+2)
on 3: 4-3=1, a[1] (+1)
    ...a[4] = 6
